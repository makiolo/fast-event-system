73 std::string EnvironmentImpl::osVersionImpl()
74 {
75  OSVERSIONINFOW vi;
76  vi.dwOSVersionInfoSize = sizeof(vi);
77  if (GetVersionExW(&vi) == 0) throw SystemException("Cannot get OS version information");
78  std::ostringstream str;
79  str << vi.dwMajorVersion << "." << vi.dwMinorVersion << " (Build " << (vi.dwBuildNumber & 0xFFFF);
80  std::string version;
81  UnicodeConverter::toUTF8(vi.szCSDVersion, version);
82  if (!version.empty()) str << ": " << version;

122 std::string EnvironmentImpl::nodeNameImpl()
123 {
124  HKEY hKey;
125  DWORD dwDisposition;
126  if (RegCreateKeyExW(HKEY_LOCAL_MACHINE, L"\\Ident", 0, 0, 0, 0, 0, &hKey, &dwDisposition) != ERROR_SUCCESS)
127  throw SystemException("Cannot get node name", "registry key not found");
128 
129  std::string value;
130  DWORD dwType;
131  BYTE bData[1026];
132  DWORD dwData = sizeof(bData);
133  if (RegQueryValueExW(hKey, L"Name", 0, &dwType, bData, &dwData) == ERROR_SUCCESS)
134  {
135  switch (dwType)
136  {
137  case REG_SZ:
138  UnicodeConverter::toUTF8(reinterpret_cast<wchar_t*>(bData), value);
139  break;
140 
141  default:
142  RegCloseKey(hKey);
143  throw SystemException("Cannot get node name", "registry value has wrong type");
144  }
145  }
146  else
147  {
148  RegCloseKey(hKey);
149  throw SystemException("Cannot get node name", "registry value not found");
150  }
151  RegCloseKey(hKey);
152  return value;
153 }

156 void EnvironmentImpl::nodeIdImpl(NodeId& id)
157 {
158  PIP_ADAPTER_INFO pAdapterInfo;
159  PIP_ADAPTER_INFO pAdapter = 0;
160  ULONG len = sizeof(IP_ADAPTER_INFO);
161  pAdapterInfo = reinterpret_cast<IP_ADAPTER_INFO*>(new char[len]);
162  // Make an initial call to GetAdaptersInfo to get
163  // the necessary size into len
164  DWORD rc = GetAdaptersInfo(pAdapterInfo, &len);
165  if (rc == ERROR_BUFFER_OVERFLOW)
166  {
167  delete [] reinterpret_cast<char*>(pAdapterInfo);
168  pAdapterInfo = reinterpret_cast<IP_ADAPTER_INFO*>(new char[len]);
169  }
170  else if (rc != ERROR_SUCCESS)
171  {
172  throw SystemException("cannot get network adapter list");
173  }
174  try
175  {
176  bool found = false;
177  if (GetAdaptersInfo(pAdapterInfo, &len) == NO_ERROR)
178  {
179  pAdapter = pAdapterInfo;
180  while (pAdapter && !found)
181  {
182  if (pAdapter->Type == MIB_IF_TYPE_ETHERNET && pAdapter->AddressLength == sizeof(id))
183  {
184  std::memcpy(&id, pAdapter->Address, pAdapter->AddressLength);
185  found = true;
186  }
187  pAdapter = pAdapter->Next;
188  }
189  }
190  else throw SystemException("cannot get network adapter list");
191  if (!found) throw SystemException("no Ethernet adapter found");
192  }
193  catch (Exception&)
194  {
195  delete [] reinterpret_cast<char*>(pAdapterInfo);
196  throw;
197  }
198  delete [] reinterpret_cast<char*>(pAdapterInfo);
199 }

210 bool EnvironmentImpl::envVar(const std::string& name, std::string* value)
211 {
212  if (icompare(name, TEMP) == 0)
213  {
214  if (value) *value = Path::temp();
215  }
216  else if (icompare(name, TMP) == 0)
217  {
218  if (value) *value = Path::temp();
219  }
220  else if (icompare(name, HOMEPATH) == 0)
221  {
222  if (value) *value = Path::home();
223  }
224  else if (icompare(name, COMPUTERNAME) == 0)
225  {
226  if (value) *value = nodeNameImpl();
227  }
228  else if (icompare(name, OS) == 0)
229  {
230  if (value) *value = osNameImpl();
231  }
232  else if (icompare(name, NUMBER_OF_PROCESSORS) == 0)
233  {
234  if (value) *value = NumberFormatter::format(processorCountImpl());
235  }
236  else if (icompare(name, PROCESSOR_ARCHITECTURE) == 0)
237  {
238  if (value) *value = osArchitectureImpl();
239  }
240  else return false;
241  return true;
242 }
